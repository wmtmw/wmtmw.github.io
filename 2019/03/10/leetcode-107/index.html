leetcodede_107

Posted on 2019-03-10

题目来源：https://leetcode.com/problems/binary-tree-level-order-traversal-ii/

思路： 按层遍历

代码：

/**

  • Definition for a binary tree node.
  • public class TreeNode {
  • int val;
  • TreeNode left;
  • TreeNode right;
  • TreeNode(int x) { val = x; }
  • }
    */
    class Solution {
    public List<List> levelOrderBottom(TreeNode root) {
    Queue<TreeNode> queue = new LinkedList<>(); //记录节点
List<List<Integer>> result = new ArrayList<>(); //记录最终结果
    //将根节点假如queue中
queue.add(root);
while (!queue.isEmpty()){
    List<Integer> currentResult = new ArrayList<>(); //记录每层所有节点的值
    int size = queue.size();  //记录该层节点总数
    for(int i = 0; i < size;i++){ //遍历该层左右节点
        TreeNode node = queue.remove(); //从队列中依次取出该层节点
        if(node != null){
            currentResult.add(node.val);  //记录当点节点的值
            //将当前节点的左右子节点加入queue中，遍历所有节点后，便得到了下一层的所有节点，因为LinkedList是链表，是有序的，所以会严格按照左右节点的顺序
            queue.add(node.left);  
            queue.add(node.right);
        }
    }
    if(!currentResult.isEmpty()){
    //将当前层的所有节点的值加入result的头部，读出时才是从子节点到根节点的顺序
        result.add(0,currentResult);
    }
}
return result;
    }
    }
